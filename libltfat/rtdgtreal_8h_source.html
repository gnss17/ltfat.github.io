101 /* \defgroup rtdgtrealprocessor Real-Time Discrete Gabor Transform Processor
102  * \addtogroup rtdgtrealprocessor
103  * @{
104  * The real-time DGT processor wraps the analysis-modify-synthesis loop for
105  * audio streams. It provides a callback interface which allows user-defined
106  * coefficient manipulation.
107  *
108  * Example:
109  * ~~~~~~~~~~~~~~~{.c}
110  * // Simple coefficient modification
111  * void process(void *userdata, const ltfat_complex_s inCoef[], const int M2, const int W, ltfat_complex_s outCoef[])
112  * {
113  * for(int w=0; w<W; w++) // Loop over channels
114  * for(int m=0; m<M2; m++) // Loop over frequencies
115  * out[m+w*M2] = 2.0f*in[m+w*M2];
116  * }
117  *
118  * // Initialize
119  * ltfat_rtdgtreal_processor_state_s* procstate = NULL;
120  * ltfat_rtdgtreal_processor_init_win_s( LTFAT_HANN, 1024, 256, 1024, maxChanNo, &process, NULL, &procstate);
121  *
122  * // In the audio loop
123  * void audioCallback(float** data, int dataLen, int chanNo)
124  * {
125  * ltfat_rtdgtreal_processor_execute_s(procstate, data, dataLen, chanNo, data);
126  * }
127  *
128  * // Teardown
129 * ltfat_rtdgtreal_processor_done_s(&procstate);
130  * ~~~~~~~~~~~~~~~
131  */
